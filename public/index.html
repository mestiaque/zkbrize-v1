6 ERP Controller Example 
// routes/api.php
Route::get('/fetch-employee',   [ZKController::class, 'fetchEmployee']);
Route::post('/sync-to-laravel', [ZKController::class, 'syncAttendance']);

// ZKController.php
public function fetchEmployee()
{
    $employees = Employee::all()->map(fn($e) => [
        'uid'         => $e->id,
        'employee_id' => $e->employee_id,
        'name'        => $e->name,
        'privilege'   => 0,
        'password'    => '',
    ]);
    return response()->json(['data' => $employees]);
}

public function syncAttendance(Request $request)
{
    $records = $request->input('records', []);
    foreach ($records as $rec) {
        Attendance::updateOrCreate(
            ['employee_id' => $rec['employee_id'], 'punch_time' => $rec['time']],
            ['status' => $rec['status'], 'device_id' => $rec['device_id'], 'source' => $rec['source']]
        );
    }
    return response()->json(['success' => true, 'saved' => count($records)]);
}
